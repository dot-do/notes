## Pattern Established

### Type Structure
```typescript
// types.ts
export interface CoreConfig { ... }
export interface ResourceDefinition<TInput, TOutput> { ... }
export interface ExecutionResult<T> { ... }
export type Status = 'pending' | 'running' | 'completed' | 'failed'
```

### Client Structure
```typescript
// client.ts
export class Client {
  private registry: Registry = {}
  register<T, U>(name: string, def: Definition<T, U>): void
  execute<T, U>(name: string, input: T): Promise<Result<U>>
  list(): Definition[]
  get(name: string): Definition | undefined
}
```

```markdown
---
title: [Domain] Reference

```
