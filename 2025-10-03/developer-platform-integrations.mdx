import { Session } from '@shopify/shopify-api'

export async function handleOAuth(shop: string, code: string) {
  const accessToken = await shopify.auth.callback({
    rawRequest: req,
    rawResponse: res
  })

  // Store session in .do database
  await env.DB.createThing({
    ns: 'shopify-sessions',
    id: shop,
    type: 'ShopifySession',
    data: {
      shop,
      accessToken,
      scope: ['read_products', 'write_orders']
    }
  })

  return accessToken
}

// shopify-app/src/webhooks.ts
export async function handleWebhook(req: Request) {
  const hmac = req.headers.get('X-Shopify-Hmac-SHA256')
  const verified = await shopify.webhooks.validate({
    rawBody: await req.text(),
    rawRequest: req,
    rawResponse: null
  })

  if (!verified) {
    return new Response('Invalid webhook', { status: 401 })
  }

  // Process webhook via .do workers
  await env.WEBHOOKS.handleShopifyWebhook(await req.json())
}

async function createAgent() {
  const name = await vscode.window.showInputBox({

    placeHolder: 'my-agent'

  const agent = await client.agents.create({ name })

  vscode.window.showInformationMessage(`Agent ${name} created!`)

**OAuth Flow:**
```typescript
// src/callback.ts
export async function handleCallback(req: VercelRequest, res: VercelResponse) {
  const { code, configurationId, teamId, next } = req.query

  // Exchange code for Vercel access token
  const tokenResponse = await fetch('https://api.vercel.com/v2/oauth/access_token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      client_id: process.env.VERCEL_CLIENT_ID,
      client_secret: process.env.VERCEL_CLIENT_SECRET,
      code,
      redirect_uri: process.env.VERCEL_REDIRECT_URI
    })
  })

  const { access_token } = await tokenResponse.json()

  // Store integration config in .do database
  await env.DB.createThing({
    ns: 'integrations',
    id: configurationId as string,
    type: 'VercelIntegration',
    data: {
      teamId,
      accessToken: access_token,
      createdAt: new Date().toISOString()
    }
  })

  // Redirect to configuration UI
  res.redirect(next as string || '/configure')
}
```

function dot_do_generate_content($prompt) {
