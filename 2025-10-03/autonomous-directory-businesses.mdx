interface APIProduct {
  id: string
  name: string
  baseUrl: string
  openApiSpec?: string
  authentication: 'api-key' | 'oauth2' | 'bearer'
  category: string[]
  description: string
  pricing: 'free' | 'freemium' | 'paid'
}

        .catch(async () => {

        })
