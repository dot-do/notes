```javascript
// triggers/new_generation.js
module.exports = {
  key: 'new_generation',
  noun: 'AI Generation',
  display: {
    label: 'New AI Generation',
    description: 'Triggers when a new AI generation completes'
  },
  operation: {
    type: 'hook',
    performSubscribe: async (z, bundle) => {
      const response = await z.request({
        url: 'https://api.do/webhooks',
        method: 'POST',
        body: {
          url: bundle.targetUrl,
          event: 'generation.completed'
        }
      })
      return response.data
    },
    performUnsubscribe: async (z, bundle) => {
      await z.request({
        url: `https://api.do/webhooks/${bundle.subscribeData.id}`,
        method: 'DELETE'
      })
    },
    perform: async (z, bundle) => {
      return [bundle.cleanedRequest]
    }
  }
}
```

**3. Actions**
