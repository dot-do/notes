#### TypeScript Example
```typescript
import { createClient } from '@dot-do/api'

// Initialize client
const client = createClient({
  apiKey: process.env.API_KEY,
  baseUrl: 'https://api.do/v1'
})

// Health check
const health = await client.health()
console.log('API Status:', health.status)

// List services
const services = await client.services.list()
console.log('Available services:', services.services.length)

// Get specific service status
const dbStatus = await client.services.status('db')
console.log('DB Service:', {
  status: dbStatus.status,
  latency: dbStatus.latency,
  uptime: dbStatus.uptime
})

// Test route resolution
const route = await client.routes.test({
  method: 'GET',
  path: '/db/users/123'
})
console.log('Route matches:', route.serviceName)

// Get metrics
const metrics = await client.metrics({ timeRange: '24h' })
console.log('Gateway metrics:', {
  totalRequests: metrics.totalRequests,
  averageLatency: metrics.averageLatency,
  successRate: (metrics.successfulRequests / metrics.totalRequests * 100).toFixed(2) + '%'
})
```

#### Python Example
```python
import requests
from datetime import datetime

# Configuration
API_KEY = 'your_api_key_here'
BASE_URL = 'https://api.do/v1'
headers = {'Authorization': f'Bearer {API_KEY}'}

# Health check
response = requests.get(f'{BASE_URL}/health')
health = response.json()
print(f"API Status: {health['status']}")

# List services
response = requests.get(f'{BASE_URL}/services', headers=headers)
services = response.json()
print(f"Available services: {len(services['services'])}")

# Get service status
response = requests.get(f'{BASE_URL}/services/db/status', headers=headers)
status = response.json()
print(f"DB Service: status={status['status']}, latency={status['latency']}ms")

# Test route resolution
payload = {
    'method': 'GET',
    'path': '/db/users/123'
}
response = requests.post(f'{BASE_URL}/routes', json=payload, headers=headers)
route = response.json()
print(f"Route matches: {route['serviceName']}")

# Get metrics
params = {'timeRange': '24h'}
response = requests.get(f'{BASE_URL}/metrics', params=params, headers=headers)
```

```sql
CREATE TABLE api_calls (

  service String,
  endpoint String,
  marketplace String,
  api_key_id String,

  latency_ms Float32,
  user_id String
) ENGINE = MergeTree()
ORDER BY (timestamp, service, marketplace)

-- Query marketplace performance
SELECT
  marketplace,
  service,
  COUNT(*) as total_calls,
  AVG(latency_ms) as avg_latency,
  COUNT(DISTINCT user_id) as unique_users,
  SUM(CASE WHEN status_code = 200 THEN 1 ELSE 0 END) / COUNT(*) * 100 as success_rate
FROM api_calls
WHERE timestamp >= now() - INTERVAL 30 DAY
GROUP BY marketplace, service
ORDER BY total_calls DESC
```
